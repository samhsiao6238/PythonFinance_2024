# 期貨測試單紀錄

_參考 [官方說明](https://sinotrade.github.io/zh_TW/tutor/prepare/example_testing_flow/)，因為官方的期貨測試單似乎有點錯誤，這裡記錄修正_

<br>

## 操作步驟

1. 編輯 `pyproject.toml` 文件中的 `[project.scripts]`，添加

    ```bash
    stock_testing = "sj_trading.testing_flow:testing_futures_ordering"
    ```

<br>

2. 紀錄 `__init__.py` 內容。

    ```python
    import shioaji as sj
    import os
    from dotenv import load_dotenv

    load_dotenv()

    def main():
        api = sj.Shioaji(simulation=True)
        api.login(
            api_key=os.environ["API_KEY"],
            secret_key=os.environ["SECRET_KEY"],
            fetch_contract=False
        )
        api.activate_ca(
            ca_path=os.environ["CA_CERT_PATH"],
            ca_passwd=os.environ["CA_PASSWORD"],
        )
        print("login and activate ca success")

    def hello():
        get_shioaji_client()

    def get_shioaji_client() -> sj.Shioaji:
        api = sj.Shioaji()
        print("Shioaji API created")
        return api
    ```

<br>

3. 在 `sec/sj_trading`添加 `testing_flow.py` 文件。

    ```python
    import shioaji as sj
    from shioaji.constant import Action, FuturesPriceType, FuturesOCType, OrderType
    import os

    def testing_futures_ordering():
        try:
            # 初始化 Shioaji API
            print("初始化 Shioaji API（模擬環境）...")
            api = sj.Shioaji(simulation=True)
            
            # 嘗試登入
            print("嘗試登入...")
            accounts = api.login(
                api_key=os.environ["API_KEY"],
                secret_key=os.environ["SECRET_KEY"],
            )
            print(f"Available accounts: {accounts}")
            
            # 啟用 CA 憑證
            print("啟用 CA 憑證...")
            api.activate_ca(
                ca_path=os.environ["CA_CERT_PATH"],
                ca_passwd=os.environ["CA_PASSWORD"],
            )
            
            # 獲取最近月的期貨合約
            print("獲取最近月的期貨合約...")
            contract = min(
                [
                    x for x in api.Contracts.Futures.TXF
                    if x.code[-2:] not in ["R1", "R2"]
                ],
                key=lambda x: x.delivery_date
            )
            print(f"Selected contract: {contract}")
            
            # 確保價格合法
            valid_price = max(contract.limit_down + 1, min(contract.limit_up - 1, contract.reference))
            print(f"Valid price for order: {valid_price}")
            
            # 建立期貨委託單
            order = api.Order(
                action=Action.Buy,
                price=valid_price,
                quantity=1,
                price_type=FuturesPriceType.LMT,
                order_type=OrderType.ROD,
                octype=FuturesOCType.Auto,
                account=api.futopt_account
            )
            print(f"Order: {order}")
            
            # 送出委託單
            trade = api.place_order(contract=contract, order=order)
            print(f"Trade: {trade}")
            
            # 更新狀態
            api.update_status()
            print(f"Status: {trade.status}")

        except Exception as e:
            print(f"發生錯誤：{e}")

        finally:
            # 確保釋放連線
            if 'api' in locals():
                print("釋放連線...")
                api.logout()
                print("已成功登出 Shioaji API。")
    ```

<br>

4. 在專案根目錄運行。

    ```bash
    uv run stock_testing
    ```
