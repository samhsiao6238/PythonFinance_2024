# 查詢當前回饋


## 程式碼

1. 完整腳本。

```python
# 導入庫
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
from selenium.webdriver.common.by import By
from selenium.webdriver import Chrome

# 設定要查詢的關鍵字
keywords = [
    '家樂福',
    '金車線上購',
    'Apple 官方網站',
    '蝦皮超市',
    'adidas 官方網站',
    'NIKE',
    '寶雅線上買',
    '松果購物',
    'PChome 24h購物',
    'LG官方線上商城',
    '東森購物',
    'Her森森購物網',
    '三立電電購'
]

# 設定瀏覽選項
options = Options()
options.add_argument("--disable-notifications")
# 新版
chrome = Chrome(options=options)

for keyword in keywords:
    URL = f"https://buy.line.me/s/{keyword}"
    chrome.get(URL)

    # 注意：以下Xpath假設對所有關鍵字都適用，實際情況可能需要調整
    try:
        xpath_name = "//div[@class='suggestedMerchantCards-title']/h2[@class='suggestedMerchantCards-titleName']"
        xpath_perc = "//div[@class='suggestedMerchantCards-point point']"
        _name = chrome.find_element(By.XPATH, xpath_name)
        _perc = chrome.find_element(By.XPATH, xpath_perc)
        print("商家名稱:", _name.text)
        print("回饋:", _perc.text)
    except Exception as e:
        print("商家名稱:", _name.text)
        print("回饋:", '當前沒有優惠或搜尋時發生錯誤')

chrome.quit()
```
