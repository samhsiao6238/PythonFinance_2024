<br>

3. 定義主類別 `Trade`。

    ```python
    # 主類別
    class Trade:
        def __init__(self, init_time, init_price, init_unit):
            # 確保初始價格存在於跳動點價格列表中
            self.stock_price_list = (
                [i / 100 for i in range(1, 1001)]
                + [10 + (i / 100) for i in range(5, 4005, 5)]
                + [50 + (i / 10) for i in range(1, 501)]
                + [100 + (i / 10) for i in range(5, 4005, 5)]
                + [500 + (i) for i in range(1, 501)]
                + [1000 + (i) for i in range(5, 4005, 5)]
            )

            # 校正初始價格
            if init_price not in self.stock_price_list:
                init_price = min(
                    [i for i in self.stock_price_list if i >= round(init_price, 2)],
                    default=init_price,
                )
                print(f"警告: 初始價格不在跳動點價格列表中，已校正為 {init_price}")

            # 初始化交易明細表
            self.position_table = pd.DataFrame(
                {
                    "進場時間": [init_time] * init_unit,
                    "進場價格": [init_price] * init_unit,
                    "出場時間": [None] * init_unit,
                    "出場價格": [None] * init_unit,
                    # 初始化持倉部位
                    "買進後持有": list(range(init_unit, 0, -1))
                }
            )

        def position(self):
            # 查詢在倉部位
            return self.position_table[
                self.position_table["出場時間"].isna()
            ].shape[0]

        def cover(self, time, price):
            # 校正價格到跳動點
            excute_price = max(
                [i for i in self.stock_price_list if i <= round(price, 2)],
                default=price,
            )
            if excute_price == price:
                print(f"警告: 平倉價格未匹配到跳動點，使用當前價格: {price}")

            for index, row in self.position_table.iterrows():
                if row["出場時間"] is None:
                    # 更新出場時間與價格
                    self.position_table.loc[index, "出場時間"] = time
                    self.position_table.loc[index, "出場價格"] = excute_price
                    # 更新 "買進後持有"
                    self.position_table.loc[index, "買進後持有"] = self.position() - 1
                    break

        def order(self, time, price):
            # 校正價格到跳動點
            filtered_prices = [
                i for i in self.stock_price_list if i >= round(price, 2)
            ]
            excute_price = min(filtered_prices, default=round(price, 2))
            if excute_price == price:
                print(f"警告: 下單價格未匹配到跳動點，使用當前價格: {price}")

            # 添加新的交易記錄
            new_trade = pd.DataFrame(
                [[
                    time, 
                    excute_price, 
                    None, 
                    None, 
                    self.position() + 1
                ]],
                columns=[
                    "進場時間",
                    "進場價格",
                    "出場時間",
                    "出場價格",
                    "買進後持有",
                ],
            )
            self.position_table = pd.concat(
                [self.position_table, new_trade], 
                ignore_index=True
            )

        def convertYearFormat(self, date):
            date1 = date.replace("年", "-").replace("月", "-").replace("日", "")
            date2 = date1.split("-")
            date3 = "-".join([
                str(int(date2[0]) + 1911), 
                date2[1], 
                date2[2]
            ])
            return date3

        def restoreReturn(self, symbol):
            dirname = "data"
            if not os.path.exists(dirname):
                os.makedirs(dirname)
            dividend_filename = f"{dirname}/上市除權息表.csv"
            if not os.path.exists(dividend_filename):
                try:
                    today = datetime.now().strftime("%Y%m%d")
                    url = f"https://www.twse.com.tw/rwd/zh/exRight/TWT49U?startDate=20080101&endDate={today}&response=json"
                    html = requests.get(url)
                    # 檢查請求是否成功
                    html.raise_for_status()
                    dividend_data = loads(html.text)
                    dividend_table = pd.DataFrame(
                        dividend_data["data"], 
                        columns=dividend_data["fields"]
                    )
                    dividend_table.to_csv(
                        dividend_filename, 
                        index=False, 
                        encoding="cp950"
                    )
                except Exception as e:
                    print(f"除權息數據下載失敗: {e}")
                    return

            # 讀取除權息數據
            dividend_table = pd.read_csv(
                dividend_filename, 
                encoding="cp950"
            )
            dividend_table["西元日期"] = dividend_table["資料日期"].apply(
                self.convertYearFormat
            )
            dividend_table_one = dividend_table[
                dividend_table["股票代號"] == symbol
            ].copy()
            dividend_table_one["西元日期"] = pd.to_datetime(
                dividend_table_one["西元日期"] + " 09:00"
            )

            # 判斷每筆交易是否有包含除權息
            self.position_table["還原除權息"] = 0.0
            for index, row in self.position_table[
                self.position_table["出場時間"].notna()
            ].iterrows():
                for _, row1 in dividend_table_one.iterrows():
                    if row["進場時間"] <= row1["西元日期"] <= row["出場時間"]:
                        self.position_table.loc[index, "還原除權息"] += float(
                            row1["權值+息值"]
                        )
    ```
